\newcommand{\plt}[4][]{

		\ifthenelse{\equal{#1}{}}{}{\vspace{1em}}
		\begin{center}
			\input{plots/#2.pgf}
		\end{center}
		\ifthenelse{\equal{#1}{}}{}{
			\begin{textblock}{2}(0.2,-2.75)
				\textbf{a.}

				\textbf{b.}

		}

	\end{figure}
}
